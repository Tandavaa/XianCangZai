![林田球：异兽魔都-5.png](/img/user/%E9%99%84%E4%BB%B6/%E9%99%84%E4%BB%B62023/%E6%9E%97%E7%94%B0%E7%90%83%EF%BC%9A%E5%BC%82%E5%85%BD%E9%AD%94%E9%83%BD-5.png)
